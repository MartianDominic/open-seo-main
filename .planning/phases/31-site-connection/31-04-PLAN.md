```typescript
import { createServerFn } from '@tanstack/react-start';
import { z } from 'zod';
import { detectPlatform } from '@/server/features/connections';
import { connectionService, type CreateConnectionInput } from '@/server/features/connections/services/ConnectionService';
import type { DetectionResult, PlatformType } from '@/server/features/connections/types';

// Schema for input validation
const detectPlatformSchema = z.object({
  domain: z.string().min(1),
});

const createConnectionSchema = z.object({
  clientId: z.string().min(1),
  platform: z.enum(['wordpress', 'shopify', 'wix', 'squarespace', 'webflow', 'custom', 'pixel']),
  siteUrl: z.string().url(),
  displayName: z.string().optional(),
  credentials: z.record(z.unknown()),
});

/**
 * Detect platform for a domain.
 */
export const detectPlatformFn = createServerFn('POST', async (input: z.infer<typeof detectPlatformSchema>) => {
  const { domain } = detectPlatformSchema.parse(input);
  return detectPlatform(domain);
});

/**
 * Create a new site connection.
 * Credentials are encrypted server-side.
 */
export const createConnectionFn = createServerFn('POST', async (input: z.infer<typeof createConnectionSchema>) => {
  const validated = createConnectionSchema.parse(input);
  return connectionService.createConnection(validated);
});

/**
 * Verify a connection and update status.
 */
export const verifyConnectionFn = createServerFn('POST', async (input: { connectionId: string }) => {
  const { connectionId } = z.object({ connectionId: z.string() }).parse(input);
  return connectionService.verifyConnection(connectionId);
});

/**
 * Get connections for a client.
 */
export const getConnectionsFn = createServerFn('GET', async (input: { clientId: string }) => {
  const { clientId } = z.object({ clientId: z.string() }).parse(input);
  return connectionService.getConnectionsForClient(clientId);
});

/**
 * Delete a connection.
 */
export const deleteConnectionFn = createServerFn('POST', async (input: { connectionId: string }) => {
  const { connectionId } = z.object({ connectionId: z.string() }).parse(input);
  await connectionService.deleteConnection(connectionId);
  return { success: true };
});
```

**ConnectionWizard.tsx:**
Multi-step wizard with states:
1. DETECT - Show PlatformSelector
2. CREDENTIALS - Show platform-specific form (WordPress/Shopify)
3. VERIFYING - Show loading spinner
4. COMPLETE - Show success with ConnectionStatus

```typescript
import { useState } from 'react';
import { PlatformSelector } from './PlatformSelector';
import { WordPressCredentialForm } from './WordPressCredentialForm';
import { ShopifyOAuthButton } from './ShopifyOAuthButton';
import { ConnectionStatus } from './ConnectionStatus';
import { createConnectionFn, verifyConnectionFn } from '@/serverFunctions/connections';
import type { PlatformType, ConnectionWithoutCredentials } from '@/server/features/connections/types';

interface ConnectionWizardProps {
  clientId: string;
  onComplete: (connection: ConnectionWithoutCredentials) => void;
  onCancel: () => void;
}

type WizardStep = 'DETECT' | 'CREDENTIALS' | 'VERIFYING' | 'COMPLETE';

export function ConnectionWizard({ clientId, onComplete, onCancel }: ConnectionWizardProps) {
  const [step, setStep] = useState<WizardStep>('DETECT');
  const [platform, setPlatform] = useState<PlatformType | null>(null);
  const [siteUrl, setSiteUrl] = useState('');
  const [connection, setConnection] = useState<ConnectionWithoutCredentials | null>(null);
  const [error, setError] = useState<string | null>(null);
  
  const handlePlatformSelected = (p: PlatformType, url: string) => {
    setPlatform(p);
    setSiteUrl(url);
    setStep('CREDENTIALS');
  };
  
  const handleCredentialsSubmit = async (credentials: Record<string, unknown>) => {
    setStep('VERIFYING');
    setError(null);
    
    try {
      // Create connection
      const conn = await createConnectionFn({
        clientId,
        platform: platform!,
        siteUrl,
        credentials,
      });
      
      // Verify connection
      const result = await verifyConnectionFn({ connectionId: conn.id });
      
      if (!result.success) {
        setError(result.error ?? 'Verification failed');
        setStep('CREDENTIALS');
        return;
      }
      
      setConnection(conn);
      setStep('COMPLETE');
    } catch (err) {
      setError((err as Error).message);
      setStep('CREDENTIALS');
    }
  };
  
  // Render based on step...
}
```

**index.tsx - Connections list page:**
```typescript
import { createFileRoute, useParams, Link } from '@tanstack/react-router';
import { useQuery } from '@tanstack/react-query';
import { getConnectionsFn } from '@/serverFunctions/connections';
import { ConnectionStatus } from '@/components/connections/ConnectionStatus';
import { Button } from '@/components/ui/button';
import { Plus } from 'lucide-react';

export const Route = createFileRoute('/clients/$clientId/connections/')({
  component: ConnectionsListPage,
});

function ConnectionsListPage() {
  const { clientId } = useParams({ from: '/clients/$clientId/connections/' });
  
  const { data: connections, isLoading } = useQuery({
    queryKey: ['connections', clientId],
    queryFn: () => getConnectionsFn({ clientId }),
  });
  
  return (
    <div className="container mx-auto py-8">
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-2xl font-bold">Site Connections</h1>
        <Button asChild>
          <Link to="/clients/$clientId/connections/new" params={{ clientId }}>
            <Plus className="w-4 h-4 mr-2" />
            Add Connection
          </Link>
        </Button>
      </div>
      
      {isLoading ? (
        <p>Loading...</p>
      ) : connections?.length === 0 ? (
        <p className="text-muted-foreground">No connections yet. Add one to get started.</p>
      ) : (
        <div className="space-y-4">
          {connections?.map((conn) => (
            <div key={conn.id} className="border rounded-lg p-4">
              <div className="flex justify-between items-start">
                <div>
                  <h3 className="font-medium">{conn.displayName ?? conn.siteUrl}</h3>
                  <p className="text-sm text-muted-foreground">{conn.platform}</p>
                </div>
                <ConnectionStatus connection={conn} />
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```
