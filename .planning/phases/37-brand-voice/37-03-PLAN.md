export class VoiceProfileService {
  async create(clientId: string, data: Partial<VoiceProfileInsert>): Promise<VoiceProfileSelect> {
    const id = nanoid();
    const [profile] = await db.insert(voiceProfiles).values({
      id,
      clientId,
      mode: data.mode ?? "best_practices",
      // ... all dimension fields with defaults
      ...data,
      createdAt: new Date(),
      updatedAt: new Date(),
    }).returning();
    return profile;
  }
  
  async createFromTemplate(
    clientId: string, 
    templateId: string, 
    overrides?: Partial<VoiceProfileInsert>
  ): Promise<VoiceProfileSelect> {
    const template = getTemplateDefaults(templateId);
    if (!template) throw new Error(`Unknown template: ${templateId}`);
    return this.create(clientId, { ...template, ...overrides });
  }
  
  async getByClientId(clientId: string): Promise<VoiceProfileSelect | null> {
    const [profile] = await db.select().from(voiceProfiles)
      .where(eq(voiceProfiles.clientId, clientId));
    return profile ?? null;
  }
  
  async update(profileId: string, data: Partial<VoiceProfileInsert>): Promise<VoiceProfileSelect> {
    const [updated] = await db.update(voiceProfiles)
      .set({ ...data, updatedAt: new Date() })
      .where(eq(voiceProfiles.id, profileId))
      .returning();
    return updated;
  }
  
  async delete(profileId: string): Promise<void> {
    await db.delete(voiceProfiles).where(eq(voiceProfiles.id, profileId));
    // FK cascade handles voice_analysis cleanup
  }
}

export const voiceProfileService = new VoiceProfileService();

export class ProtectionRulesService {
  async create(profileId: string, data: {
    rule_type: 'page' | 'section' | 'pattern';
    target: string;
    reason: string;
    created_by: string;
    expires_at?: Date;
  }): Promise<ContentProtectionRuleSelect> {
    // Validate target based on rule_type
    this.validateTarget(data.rule_type, data.target);
    
    const id = nanoid();
    const [rule] = await db.insert(contentProtectionRules).values({
      id, profileId, ...data, createdAt: new Date()
    }).returning();
    return rule;
  }
  
  validateTarget(type: string, target: string): void {
    if (type === 'section') {
      // Validate CSS selector syntax
      try { document.querySelector(target); } catch { throw new Error('Invalid CSS selector'); }
    } else if (type === 'pattern') {
      // Validate regex
      try { new RegExp(target); } catch { throw new Error('Invalid regex pattern'); }
    } else if (type === 'page') {
      // Validate URL format
      try { new URL(target); } catch { throw new Error('Invalid URL'); }
    }
  }
  
  async getByProfileId(profileId: string): Promise<ContentProtectionRuleSelect[]> {
    return db.select().from(contentProtectionRules)
      .where(eq(contentProtectionRules.profileId, profileId));
  }
  
  async getActiveRules(profileId: string): Promise<ContentProtectionRuleSelect[]> {
    return db.select().from(contentProtectionRules)
