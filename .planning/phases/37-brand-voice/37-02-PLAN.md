```typescript
export default async function processor(job: Job<VoiceAnalysisJobData>) {
  const { clientId, profileId, urls, progress } = job.data;
  const startIdx = progress?.completedUrls ?? 0;
  
  // Scrape pages using existing multiPageScraper
  // For each page: extract voice with VoiceAnalyzer
  // Save each result to voice_analysis table
  // Update job progress via job.updateProgress()
  // On completion: aggregate results, update voice_profiles
  
  // Checkpoint pattern: save progress in job.updateData()
  for (let i = startIdx; i < urls.length; i++) {
    const page = await scrapeUrl(urls[i]);
    const result = await analyzePageVoice(page, new URL(urls[i]).hostname);
    await saveVoiceAnalysis(profileId, urls[i], result);
    await job.updateData({ ...job.data, progress: { completedUrls: i + 1, totalUrls: urls.length } });
    await job.updateProgress((i + 1) / urls.length * 100);
  }
  
  // Final aggregation
  const allResults = await getVoiceAnalysisResults(profileId);
  const aggregated = aggregateVoiceResults(allResults);
  await updateVoiceProfile(profileId, aggregated);
}
```
