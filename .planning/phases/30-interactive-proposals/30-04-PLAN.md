// server/features/proposals/tracking.ts
import { db } from "@/db";
import { proposalViews } from "@/db/schema/proposals";
import { createHash } from "crypto";

interface ViewTrackingInput {
  proposalId: string;
  deviceType: "mobile" | "desktop" | "tablet";
  ipAddress: string;
  userAgent: string;
}

export async function trackProposalView(input: ViewTrackingInput) {
  const ipHash = createHash("sha256")
    .update(input.ipAddress + process.env.IP_SALT)
    .digest("hex")
    .slice(0, 16);

  await db.insert(proposalViews).values({
    proposalId: input.proposalId,
    deviceType: input.deviceType,
    ipHash,
    sectionsViewed: [],
    durationSeconds: 0,
    roiCalculatorUsed: false,
  });

  // Update first_viewed_at on proposal if not set
  await db
    .update(proposals)
    .set({ firstViewedAt: new Date(), status: "viewed" })
    .where(
      and(
        eq(proposals.id, input.proposalId),
        isNull(proposals.firstViewedAt)
      )
    );
}

export async function updateViewDuration(viewId: string, durationSeconds: number) {
  await db
    .update(proposalViews)
    .set({ durationSeconds })
    .where(eq(proposalViews.id, viewId));
}

export async function updateSectionsViewed(viewId: string, sections: string[]) {
  await db
    .update(proposalViews)
    .set({ sectionsViewed: sections })
    .where(eq(proposalViews.id, viewId));
}

export async function markRoiCalculatorUsed(viewId: string) {
  await db
    .update(proposalViews)
    .set({ roiCalculatorUsed: true })
    .where(eq(proposalViews.id, viewId));
}

// server/features/proposals/signals.ts
interface EngagementSignals {
  hot: boolean;              // 3+ views in 24h
  pricingFocused: boolean;   // Viewed pricing section 3+ times
  calculatedRoi: boolean;    // Used ROI calculator
  readyToClose: boolean;     // Multiple CTA section visits
  score: number;             // 0-100 engagement score
}

export async function calculateEngagementSignals(
  proposalId: string
): Promise<EngagementSignals> {
  const views = await db
    .select()
    .from(proposalViews)
    .where(eq(proposalViews.proposalId, proposalId))
    .orderBy(desc(proposalViews.viewedAt));

  const last24h = views.filter(
    (v) => new Date(v.viewedAt) > new Date(Date.now() - 24 * 60 * 60 * 1000)
  );

  const pricingSectionViews = views.filter(
    (v) => v.sectionsViewed?.includes("investment")
  ).length;

  const ctaSectionViews = views.filter(
    (v) => v.sectionsViewed?.includes("cta")
  ).length;

  const roiCalculatorUsed = views.some((v) => v.roiCalculatorUsed);

  // Score calculation
  let score = 0;
  score += Math.min(views.length * 10, 30);        // Up to 30 for views
  score += Math.min(last24h.length * 15, 30);      // Up to 30 for recency
  score += roiCalculatorUsed ? 20 : 0;              // 20 for calculator use
  score += Math.min(pricingSectionViews * 5, 20);  // Up to 20 for pricing focus

  return {
    hot: last24h.length >= 3,
    pricingFocused: pricingSectionViews >= 3,
    calculatedRoi: roiCalculatorUsed,
    readyToClose: ctaSectionViews >= 2 && pricingSectionViews >= 2,
    score: Math.min(score, 100),
  };
}

// Client-side tracking hook
// hooks/useProposalTracking.ts
"use client";

import { useEffect, useRef, useCallback } from "react";

export function useProposalTracking(proposalId: string) {
  const viewIdRef = useRef<string | null>(null);
  const sectionsRef = useRef<Set<string>>(new Set());
  const startTimeRef = useRef<number>(Date.now());

  // Track page view on mount
  useEffect(() => {
    fetch("/api/proposals/track/view", {
      method: "POST",
      body: JSON.stringify({ proposalId }),
    })
      .then((r) => r.json())
      .then((data) => {
        viewIdRef.current = data.viewId;
      });

    // Heartbeat for duration
    const heartbeat = setInterval(() => {
      if (viewIdRef.current) {
        const duration = Math.floor((Date.now() - startTimeRef.current) / 1000);
        fetch("/api/proposals/track/duration", {
          method: "POST",
          body: JSON.stringify({ 
            viewId: viewIdRef.current, 
            durationSeconds: duration 
          }),
        });
      }
    }, 30000);

    return () => clearInterval(heartbeat);
  }, [proposalId]);

  // Track section view
  const trackSection = useCallback((sectionId: string) => {
    sectionsRef.current.add(sectionId);
    
    if (viewIdRef.current) {
      fetch("/api/proposals/track/sections", {
        method: "POST",
        body: JSON.stringify({
          viewId: viewIdRef.current,
          sections: Array.from(sectionsRef.current),
        }),
      });
    }
  }, []);

  // Track ROI calculator use
  const trackRoiCalculator = useCallback(() => {
    if (viewIdRef.current) {
      fetch("/api/proposals/track/roi-calculator", {
        method: "POST",
        body: JSON.stringify({ viewId: viewIdRef.current }),
      });
    }
  }, []);

  return { trackSection, trackRoiCalculator };
}
