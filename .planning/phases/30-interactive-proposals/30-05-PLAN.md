```typescript
// server/lib/dokobit/client.ts
const DOKOBIT_BASE = "https://api.dokobit.com";

interface DokobitClient {
  initiateSmartIdSigning(params: SmartIdParams): Promise<SmartIdSession>;
  initiateMobileIdSigning(params: MobileIdParams): Promise<MobileIdSession>;
  getSigningStatus(sessionId: string): Promise<SigningStatus>;
  downloadSignedDocument(sessionId: string): Promise<Buffer>;
}

interface SmartIdParams {
  personalCode: string;  // Lithuanian personal code (11 digits)
  country: "LT" | "EE" | "LV";
  documentHash: string;  // SHA256 hash of PDF
  documentName: string;
}

interface SmartIdSession {
  sessionId: string;
  verificationCode: string;  // 4-digit code to show user
}

interface SigningStatus {
  status: "pending" | "completed" | "failed" | "expired";
  signedDocumentUrl?: string;
  error?: string;
}

export function createDokobitClient(): DokobitClient {
  const accessToken = process.env.DOKOBIT_ACCESS_TOKEN;
  
  if (!accessToken) {
    throw new Error("DOKOBIT_ACCESS_TOKEN not configured");
  }

  return {
    async initiateSmartIdSigning(params) {
      const response = await fetch(`${DOKOBIT_BASE}/signing/v2/smartid/sign`, {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${accessToken}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          pno: `${params.country}-${params.personalCode}`,
          hash: params.documentHash,
          hash_type: "SHA256",
          message: `Pasirašyti: ${params.documentName}`,
        }),
      });

      if (!response.ok) {
        throw new Error(`Dokobit error: ${await response.text()}`);
      }

      const data = await response.json();
      return {
        sessionId: data.session_id,
        verificationCode: data.verification_code,
      };
    },

    async initiateMobileIdSigning(params) {
      const response = await fetch(`${DOKOBIT_BASE}/signing/v2/mobileid/sign`, {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${accessToken}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          phone: params.phoneNumber,
          pno: `${params.country}-${params.personalCode}`,
          hash: params.documentHash,
          hash_type: "SHA256",
          message: `Pasirašyti: ${params.documentName}`,
        }),
      });

      if (!response.ok) {
        throw new Error(`Dokobit error: ${await response.text()}`);
      }

      const data = await response.json();
      return {
        sessionId: data.session_id,
        verificationCode: data.verification_code,
      };
    },

    async getSigningStatus(sessionId) {
      const response = await fetch(
        `${DOKOBIT_BASE}/signing/v2/session/${sessionId}/status`,
        {
          headers: { "Authorization": `Bearer ${accessToken}` },
        }
      );

      const data = await response.json();
      return {
        status: data.status,
        signedDocumentUrl: data.signed_document_url,
        error: data.error_message,
      };
    },

    async downloadSignedDocument(sessionId) {
      const response = await fetch(
        `${DOKOBIT_BASE}/signing/v2/session/${sessionId}/download`,
        {
          headers: { "Authorization": `Bearer ${accessToken}` },
        }
      );

      return Buffer.from(await response.arrayBuffer());
    },
  };
}
```

```typescript
// server/features/proposals/signing.ts
import { createDokobitClient } from "@/server/lib/dokobit/client";
import { generateContractPdf } from "./pdf";
import { uploadToR2 } from "@/server/lib/r2";
import { db } from "@/db";
import { proposalSignatures, proposals } from "@/db/schema/proposals";
import { createHash } from "crypto";

export async function initiateProposalSigning(
  proposalId: string,
  method: "smart_id" | "mobile_id",
  personalCode: string,
  phoneNumber?: string
): Promise<{ sessionId: string; verificationCode: string }> {
  const proposal = await db.query.proposals.findFirst({
    where: eq(proposals.id, proposalId),
  });

  if (!proposal) {
    throw new Error("Proposal not found");
  }

  // Generate contract PDF
  const pdfBuffer = await generateContractPdf(proposal);
  const documentHash = createHash("sha256").update(pdfBuffer).digest("hex");

  const dokobit = createDokobitClient();

  const session = method === "smart_id"
    ? await dokobit.initiateSmartIdSigning({
        personalCode,
        country: "LT",
        documentHash,
        documentName: `SEO Sutartis - ${proposal.content.domain}`,
      })
    : await dokobit.initiateMobileIdSigning({
        personalCode,
        phoneNumber: phoneNumber!,
        country: "LT",
        documentHash,
        documentName: `SEO Sutartis - ${proposal.content.domain}`,
      });

  // Store signature record
  await db.insert(proposalSignatures).values({
    proposalId,
    signerPersonalCodeHash: createHash("sha256")
      .update(personalCode + process.env.PERSONAL_CODE_SALT)
      .digest("hex"),
    signingMethod: method,
    dokobitSessionId: session.sessionId,
  });

  return session;
}

export async function checkSigningStatus(
  proposalId: string,
  sessionId: string
): Promise<SigningStatus> {
  const dokobit = createDokobitClient();
  const status = await dokobit.getSigningStatus(sessionId);

  if (status.status === "completed") {
    // Download signed PDF
    const signedPdf = await dokobit.downloadSignedDocument(sessionId);
    
    // Upload to R2
    const pdfUrl = await uploadToR2(
      `proposals/${proposalId}/signed-contract.pdf`,
      signedPdf,
      "application/pdf"
    );

    // Update records
    await db
      .update(proposalSignatures)
      .set({
        signedPdfUrl: pdfUrl,
        signedAt: new Date(),
      })
      .where(eq(proposalSignatures.dokobitSessionId, sessionId));

    await db
      .update(proposals)
      .set({
        status: "signed",
        signedAt: new Date(),
      })
      .where(eq(proposals.id, proposalId));
  }

  return status;
}
```

## Signing UI

```typescript
// components/proposals/SigningModal.tsx
"use client";

import { useState } from "react";
import { Dialog, DialogContent, DialogHeader } from "@tevero/ui";

type SigningStep = "method" | "input" | "verification" | "success" | "error";

export function SigningModal({ 
  proposalId, 
  onSuccess 
}: { 
  proposalId: string; 
  onSuccess: () => void;
}) {
  const [step, setStep] = useState<SigningStep>("method");
  const [method, setMethod] = useState<"smart_id" | "mobile_id">();
  const [personalCode, setPersonalCode] = useState("");
  const [verificationCode, setVerificationCode] = useState("");
  const [sessionId, setSessionId] = useState("");
  const [error, setError] = useState("");

  const initiateSignimg = async () => {
    const res = await fetch("/api/proposals/sign/initiate", {
      method: "POST",
      body: JSON.stringify({ proposalId, method, personalCode }),
    });
    
    if (!res.ok) {
      setError("Nepavyko inicijuoti pasirašymo");
      setStep("error");
      return;
    }

    const data = await res.json();
    setSessionId(data.sessionId);
    setVerificationCode(data.verificationCode);
    setStep("verification");

    // Start polling
    pollStatus(data.sessionId);
  };

  const pollStatus = async (sid: string) => {
    const poll = async () => {
      const res = await fetch(`/api/proposals/sign/status?sessionId=${sid}`);
      const data = await res.json();

      if (data.status === "completed") {
        setStep("success");
        onSuccess();
      } else if (data.status === "failed" || data.status === "expired") {
        setError(data.error || "Pasirašymas nepavyko");
        setStep("error");
      } else {
        // Keep polling
        setTimeout(() => poll(), 2000);
      }
    };
    poll();
  };

  return (
    <Dialog open>
      <DialogContent>
        {step === "method" && (
          <div className="space-y-4">
            <DialogHeader>Pasirašykite sutartį</DialogHeader>
            <div className="grid grid-cols-2 gap-4">
              <button
                onClick={() => { setMethod("smart_id"); setStep("input"); }}
                className="p-4 border rounded-lg hover:bg-gray-50"
              >
                <img src="/smart-id-logo.svg" className="h-8 mx-auto mb-2" />
                <span>Smart-ID</span>
              </button>
              <button
                onClick={() => { setMethod("mobile_id"); setStep("input"); }}
                className="p-4 border rounded-lg hover:bg-gray-50"
              >
                <img src="/mobile-id-logo.svg" className="h-8 mx-auto mb-2" />
                <span>Mobile-ID</span>
              </button>
            </div>
          </div>
        )}

        {step === "input" && (
          <div className="space-y-4">
            <DialogHeader>Įveskite asmens kodą</DialogHeader>
            <input
              type="text"
              placeholder="38501010001"
              value={personalCode}
              onChange={(e) => setPersonalCode(e.target.value)}
              className="w-full p-3 border rounded-lg"
              maxLength={11}
            />
            <button
              onClick={initiateSignimg}
              disabled={personalCode.length !== 11}
              className="w-full p-3 bg-primary text-white rounded-lg"
            >
              Tęsti →
            </button>
          </div>
        )}

        {step === "verification" && (
          <div className="space-y-4 text-center">
            <DialogHeader>Patvirtinimo kodas</DialogHeader>
            <div className="text-5xl font-mono tracking-widest py-8">
              {verificationCode}
            </div>
            <p className="text-muted-foreground">
              Patvirtinkite {method === "smart_id" ? "Smart-ID" : "Mobile-ID"} programėlėje
            </p>
            <div className="animate-spin w-8 h-8 border-4 border-primary border-t-transparent rounded-full mx-auto" />
          </div>
        )}

        {step === "success" && (
          <div className="text-center py-8">
            <div className="text-6xl mb-4">✓</div>
            <h3 className="text-xl font-semibold">Pasirašyta!</h3>
            <p className="text-muted-foreground">Sutartis sėkmingai pasirašyta</p>
          </div>
        )}

        {step === "error" && (
          <div className="text-center py-8">
            <div className="text-6xl mb-4">✕</div>
            <h3 className="text-xl font-semibold text-red-600">Klaida</h3>
            <p className="text-muted-foreground">{error}</p>
            <button
              onClick={() => setStep("method")}
              className="mt-4 px-4 py-2 border rounded-lg"
            >
              Bandyti dar kartą
            </button>
          </div>
        )}
      </DialogContent>
    </Dialog>
  );
}
```
