```typescript
// server/features/proposals/onboarding.ts
import { db } from "@/db";
import { clients, projects, prospects, proposals } from "@/db/schema";
import { sendEmail } from "@/server/lib/email";
import { notifyAgency } from "@/server/lib/notifications";

interface OnboardingResult {
  clientId: string;
  projectId: string;
  gscInviteSent: boolean;
  kickoffEmailSent: boolean;
}

export async function triggerOnboarding(
  proposalId: string
): Promise<OnboardingResult> {
  const proposal = await db.query.proposals.findFirst({
    where: eq(proposals.id, proposalId),
    with: {
      prospect: {
        with: {
          analyses: {
            orderBy: desc(analyses.createdAt),
            limit: 1,
          },
        },
      },
    },
  });

  if (!proposal || !proposal.prospect) {
    throw new Error("Proposal or prospect not found");
  }

  const prospect = proposal.prospect;
  const latestAnalysis = prospect.analyses[0];

  // 1. Create client
  const [client] = await db
    .insert(clients)
    .values({
      workspaceId: proposal.workspaceId,
      name: prospect.companyName || prospect.domain,
      domain: prospect.domain,
      contactEmail: prospect.contactEmail,
      contactName: prospect.contactName,
      industry: prospect.industry,
      status: "onboarding",
      convertedFromProspectId: prospect.id,
    })
    .returning();

  // 2. Update prospect status
  await db
    .update(prospects)
    .set({
      status: "converted",
      convertedClientId: client.id,
    })
    .where(eq(prospects.id, prospect.id));

  // 3. Create project
  const [project] = await db
    .insert(projects)
    .values({
      clientId: client.id,
      workspaceId: proposal.workspaceId,
      name: `SEO - ${prospect.domain}`,
      domain: prospect.domain,
      status: "setup",
      
      // Import from analysis
      targetKeywords: latestAnalysis?.organicKeywords?.slice(0, 20).map(k => k.keyword) || [],
      baselineMetrics: {
        traffic: latestAnalysis?.domainMetrics?.organicTraffic || 0,
        keywords: latestAnalysis?.domainMetrics?.organicKeywords || 0,
        domainRank: latestAnalysis?.domainMetrics?.domainRank || 0,
      },
    })
    .returning();

  // 4. Send GSC invite email
  await sendEmail({
    to: prospect.contactEmail!,
    template: "gsc-invite",
    data: {
      clientName: prospect.contactName || prospect.companyName,
      domain: prospect.domain,
      connectUrl: `${process.env.APP_URL}/onboarding/${client.id}/gsc`,
    },
  });

  // 5. Send kickoff scheduling email
  await sendEmail({
    to: prospect.contactEmail!,
    template: "kickoff-scheduling",
    data: {
      clientName: prospect.contactName || prospect.companyName,
      calendlyUrl: `${process.env.CALENDLY_URL}?email=${encodeURIComponent(prospect.contactEmail!)}`,
    },
  });

  // 6. Notify agency
  await notifyAgency({
    type: "new_client",
    data: {
      clientName: prospect.companyName || prospect.domain,
      domain: prospect.domain,
      monthlyValue: proposal.monthlyFeeCents! / 100,
      projectId: project.id,
    },
  });

  // 7. Update proposal status
  await db
    .update(proposals)
    .set({ status: "onboarded" })
    .where(eq(proposals.id, proposalId));

  return {
    clientId: client.id,
    projectId: project.id,
    gscInviteSent: true,
    kickoffEmailSent: true,
  };
}
```

```typescript
// app/api/gsc/oauth/start/route.ts
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const clientId = searchParams.get("clientId");
  
  const oauth2Client = new google.auth.OAuth2(
    process.env.GOOGLE_CLIENT_ID,
    process.env.GOOGLE_CLIENT_SECRET,
    `${process.env.APP_URL}/api/gsc/oauth/callback`
  );

  const authUrl = oauth2Client.generateAuthUrl({
    access_type: "offline",
    scope: [
      "https://www.googleapis.com/auth/webmasters.readonly",
    ],
    state: clientId!,
  });

  return Response.redirect(authUrl);
}

// app/api/gsc/oauth/callback/route.ts
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const code = searchParams.get("code");
  const clientId = searchParams.get("state");

  const oauth2Client = new google.auth.OAuth2(
    process.env.GOOGLE_CLIENT_ID,
    process.env.GOOGLE_CLIENT_SECRET,
    `${process.env.APP_URL}/api/gsc/oauth/callback`
  );

  const { tokens } = await oauth2Client.getToken(code!);

  // Store tokens for client
  await db
    .update(clients)
    .set({
      gscRefreshToken: tokens.refresh_token,
      gscConnectedAt: new Date(),
    })
    .where(eq(clients.id, clientId!));

  return Response.redirect(
    `${process.env.APP_URL}/onboarding/${clientId}/gsc/success`
  );
}
```
