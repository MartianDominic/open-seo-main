# Plan 30-01: Proposal Schema & Builder

## Goal

Database schema and admin UI for creating/editing proposals with template system.

## Deliverables

1. **Database Schema**
   - `proposals` table with content JSONB, pricing, status state machine
   - `proposal_views` table for engagement tracking  
   - `proposal_signatures` table for Dokobit integration
   - `proposal_payments` table for Stripe integration

2. **Builder UI**
   - Template selection (standard, premium, enterprise)
   - Pricing configuration (setup fee, monthly fee)
   - Brand config (logo, colors)
   - Live preview
   - Send/share functionality

## Schema

```typescript
// db/schema/proposals.ts
import { pgTable, uuid, text, integer, timestamp, jsonb, boolean } from "drizzle-orm/pg-core";

export const proposalStatusEnum = pgEnum("proposal_status", [
  "draft",
  "sent", 
  "viewed",
  "accepted",
  "signed",
  "paid",
  "onboarded",
  "expired",
  "declined"
]);

export const proposals = pgTable("proposals", {
  id: uuid("id").primaryKey().defaultRandom(),
  prospectId: uuid("prospect_id").references(() => prospects.id),
  workspaceId: uuid("workspace_id").notNull(),
  
  // Template & content
  template: text("template").default("standard"),
  content: jsonb("content").$type<ProposalContent>().notNull(),
  brandConfig: jsonb("brand_config").$type<BrandConfig>(),
  
  // Pricing
  setupFeeCents: integer("setup_fee_cents"),
  monthlyFeeCents: integer("monthly_fee_cents"),
  currency: text("currency").default("EUR"),
  
  // Status
  status: proposalStatusEnum("status").default("draft"),
  
  // Access
  token: text("token").unique().notNull(),
  expiresAt: timestamp("expires_at"),
  
  // Timestamps
  sentAt: timestamp("sent_at"),
  firstViewedAt: timestamp("first_viewed_at"),
  acceptedAt: timestamp("accepted_at"),
  signedAt: timestamp("signed_at"),
  paidAt: timestamp("paid_at"),
  
  createdAt: timestamp("created_at").defaultNow(),
  updatedAt: timestamp("updated_at").defaultNow(),
});

export const proposalViews = pgTable("proposal_views", {
  id: uuid("id").primaryKey().defaultRandom(),
  proposalId: uuid("proposal_id").references(() => proposals.id),
  
  viewedAt: timestamp("viewed_at").defaultNow(),
  durationSeconds: integer("duration_seconds"),
  sectionsViewed: text("sections_viewed").array(),
  roiCalculatorUsed: boolean("roi_calculator_used").default(false),
  deviceType: text("device_type"),
  ipHash: text("ip_hash"),
  
  createdAt: timestamp("created_at").defaultNow(),
});

export const proposalSignatures = pgTable("proposal_signatures", {
  id: uuid("id").primaryKey().defaultRandom(),
  proposalId: uuid("proposal_id").references(() => proposals.id),
  
  signerName: text("signer_name").notNull(),
  signerPersonalCodeHash: text("signer_personal_code_hash"),
  signingMethod: text("signing_method"), // 'smart_id', 'mobile_id'
  dokobitSessionId: text("dokobit_session_id"),
  signedPdfUrl: text("signed_pdf_url"),
  
  signedAt: timestamp("signed_at"),
  createdAt: timestamp("created_at").defaultNow(),
});

export const proposalPayments = pgTable("proposal_payments", {
  id: uuid("id").primaryKey().defaultRandom(),
  proposalId: uuid("proposal_id").references(() => proposals.id),
  
  provider: text("provider").default("stripe"),
  stripeSessionId: text("stripe_session_id"),
  stripePaymentIntentId: text("stripe_payment_intent_id"),
  stripeSubscriptionId: text("stripe_subscription_id"),
  
  amountCents: integer("amount_cents"),
  currency: text("currency").default("EUR"),
  status: text("status").default("pending"), // pending, completed, failed
  
  paidAt: timestamp("paid_at"),
  createdAt: timestamp("created_at").defaultNow(),
});
```
