```typescript
// components/AnimatedCounter.tsx
"use client";

import { useEffect, useRef, useState } from "react";
import { useInView } from "framer-motion";

interface AnimatedCounterProps {
  value: number;
  prefix?: string;
  suffix?: string;
  duration?: number;
}

export function AnimatedCounter({ 
  value, 
  prefix = "", 
  suffix = "",
  duration = 2000 
}: AnimatedCounterProps) {
  const ref = useRef<HTMLSpanElement>(null);
  const isInView = useInView(ref, { once: true });
  const [displayValue, setDisplayValue] = useState(0);

  useEffect(() => {
    if (!isInView) return;

    const startTime = Date.now();
    const animate = () => {
      const elapsed = Date.now() - startTime;
      const progress = Math.min(elapsed / duration, 1);
      const eased = 1 - Math.pow(1 - progress, 3); // easeOutCubic
      
      setDisplayValue(Math.floor(value * eased));
      
      if (progress < 1) {
        requestAnimationFrame(animate);
      }
    };
    
    requestAnimationFrame(animate);
  }, [isInView, value, duration]);

  return (
    <span ref={ref}>
      {prefix}{displayValue.toLocaleString()}{suffix}
    </span>
  );
}
```

```typescript
// hooks/useRoiCalculator.ts
"use client";

import { useState, useMemo } from "react";

interface RoiInputs {
  trafficGain: number;      // From analysis
  trafficValue: number;     // From analysis
  conversionRate: number;   // User adjustable, default 2%
  averageOrderValue: number; // User adjustable
  monthlyFee: number;       // From proposal
}

export function useRoiCalculator(initialData: {
  trafficGain: number;
  trafficValue: number;
  monthlyFee: number;
  defaultConversionRate?: number;
  defaultAov?: number;
}) {
  const [conversionRate, setConversionRate] = useState(
    initialData.defaultConversionRate ?? 2
  );
  const [aov, setAov] = useState(initialData.defaultAov ?? 150);

  const calculations = useMemo(() => {
    const monthlyClicks = initialData.trafficGain;
    const monthlyConversions = monthlyClicks * (conversionRate / 100);
    const monthlyRevenue = monthlyConversions * aov;
    const roi = ((monthlyRevenue - initialData.monthlyFee) / initialData.monthlyFee) * 100;
    const paybackMonths = initialData.monthlyFee / (monthlyRevenue / 12);

    return {
      trafficValue: initialData.trafficValue,
      projectedConversions: Math.round(monthlyConversions),
      projectedRevenue: Math.round(monthlyRevenue),
      roi: Math.round(roi),
      paybackMonths: paybackMonths.toFixed(1),
    };
  }, [conversionRate, aov, initialData]);

  return {
    conversionRate,
    setConversionRate,
    aov,
    setAov,
    calculations,
    resetToDefaults: () => {
      setConversionRate(initialData.defaultConversionRate ?? 2);
      setAov(initialData.defaultAov ?? 150);
    },
  };
}
```
