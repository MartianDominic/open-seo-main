```typescript
// server/features/proposals/payment.ts
import { stripe } from "@/server/lib/stripe/client";
import { db } from "@/db";
import { proposals, proposalPayments } from "@/db/schema/proposals";

interface CreateCheckoutParams {
  proposalId: string;
  customerEmail: string;
  setupFeeCents: number;
  monthlyFeeCents?: number;  // If provided, creates subscription
  successUrl: string;
  cancelUrl: string;
}

export async function createPaymentCheckout(
  params: CreateCheckoutParams
): Promise<{ checkoutUrl: string }> {
  const lineItems: Stripe.Checkout.SessionCreateParams.LineItem[] = [];

  // Setup fee (one-time)
  if (params.setupFeeCents > 0) {
    lineItems.push({
      price_data: {
        currency: "eur",
        product_data: {
          name: "SEO Įdiegimo mokestis",
          description: "Vienkartinis įdiegimo mokestis",
        },
        unit_amount: params.setupFeeCents,
      },
      quantity: 1,
    });
  }

  // Monthly fee (subscription) - if provided
  if (params.monthlyFeeCents && params.monthlyFeeCents > 0) {
    lineItems.push({
      price_data: {
        currency: "eur",
        product_data: {
          name: "SEO Mėnesinis mokestis",
          description: "Mėnesinis SEO paslaugų mokestis",
        },
        unit_amount: params.monthlyFeeCents,
        recurring: {
          interval: "month",
        },
      },
      quantity: 1,
    });
  }

  const session = await stripe.checkout.sessions.create({
    payment_method_types: ["card"],
    mode: params.monthlyFeeCents ? "subscription" : "payment",
    line_items: lineItems,
    customer_email: params.customerEmail,
    success_url: params.successUrl,
    cancel_url: params.cancelUrl,
    metadata: {
      proposalId: params.proposalId,
    },
    locale: "lt",  // Lithuanian
  });

  // Store payment record
  await db.insert(proposalPayments).values({
    proposalId: params.proposalId,
    provider: "stripe",
    stripeSessionId: session.id,
    amountCents: params.setupFeeCents + (params.monthlyFeeCents || 0),
    currency: "EUR",
    status: "pending",
  });

  return { checkoutUrl: session.url! };
}

export async function handleStripeWebhook(
  event: Stripe.Event
): Promise<void> {
  switch (event.type) {
    case "checkout.session.completed": {
      const session = event.data.object as Stripe.Checkout.Session;
      const proposalId = session.metadata?.proposalId;

      if (!proposalId) {
        console.error("No proposalId in session metadata");
        return;
      }

      // Update payment record
      await db
        .update(proposalPayments)
        .set({
          stripePaymentIntentId: session.payment_intent as string,
          stripeSubscriptionId: session.subscription as string,
          status: "completed",
          paidAt: new Date(),
        })
        .where(eq(proposalPayments.stripeSessionId, session.id));

      // Update proposal status
      await db
        .update(proposals)
        .set({
          status: "paid",
          paidAt: new Date(),
        })
        .where(eq(proposals.id, proposalId));

      // Trigger onboarding (next plan)
      await triggerOnboarding(proposalId);
      break;
    }

    case "invoice.paid": {
      // Handle subscription renewals
      const invoice = event.data.object as Stripe.Invoice;
      console.log("Invoice paid:", invoice.id);
      break;
    }

    case "customer.subscription.deleted": {
      // Handle subscription cancellation
      const subscription = event.data.object as Stripe.Subscription;
      console.log("Subscription cancelled:", subscription.id);
      break;
    }
  }
}
```

```typescript
// app/api/stripe/webhook/route.ts
import { stripe } from "@/server/lib/stripe/client";
import { handleStripeWebhook } from "@/server/features/proposals/payment";

export async function POST(request: Request) {
  const body = await request.text();
  const signature = request.headers.get("stripe-signature")!;

  let event: Stripe.Event;

  try {
    event = stripe.webhooks.constructEvent(
      body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET!
    );
  } catch (err) {
    console.error("Webhook signature verification failed");
    return new Response("Invalid signature", { status: 400 });
  }

  await handleStripeWebhook(event);

  return new Response("OK", { status: 200 });
}
```

## API Routes

```typescript
// app/api/proposals/payment/create/route.ts
import { createPaymentCheckout } from "@/server/features/proposals/payment";
import { db } from "@/db";
import { proposals } from "@/db/schema/proposals";

export async function POST(request: Request) {
  const { proposalId } = await request.json();
  
  const proposal = await db.query.proposals.findFirst({
    where: eq(proposals.id, proposalId),
    with: { prospect: true },
  });

  if (!proposal) {
    return Response.json({ error: "Proposal not found" }, { status: 404 });
  }

  const result = await createPaymentCheckout({
    proposalId,
    customerEmail: proposal.prospect?.contactEmail || "",
    setupFeeCents: proposal.setupFeeCents || 0,
    monthlyFeeCents: proposal.monthlyFeeCents,
    successUrl: `${process.env.APP_URL}/p/${proposal.token}/payment/success`,
    cancelUrl: `${process.env.APP_URL}/p/${proposal.token}?payment=cancelled`,
  });

  return Response.json(result);
}
```
