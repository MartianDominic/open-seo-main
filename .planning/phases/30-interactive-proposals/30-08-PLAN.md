## Implementation

```typescript
// server/features/proposals/automation.ts
import { db } from "@/db";
import { proposals, proposalAutomationLogs } from "@/db/schema";
import { sendEmail } from "@/server/lib/email";
import { notifyAgency } from "@/server/lib/notifications";

export async function processAutomations() {
  const rules = await db.query.automationRules.findMany({
    where: eq(automationRules.enabled, true),
  });

  for (const rule of rules) {
    const proposals = await findMatchingProposals(rule);
    
    for (const proposal of proposals) {
      // Check if already executed for this proposal
      const existing = await db.query.proposalAutomationLogs.findFirst({
        where: and(
          eq(proposalAutomationLogs.proposalId, proposal.id),
          eq(proposalAutomationLogs.ruleId, rule.id)
        ),
      });

      if (existing) continue;

      // Execute action
      await executeAction(rule.action, proposal);

      // Log execution
      await db.insert(proposalAutomationLogs).values({
        proposalId: proposal.id,
        ruleId: rule.id,
        executedAt: new Date(),
      });
    }
  }
}

async function findMatchingProposals(rule: AutomationRule) {
  if (rule.trigger.type === "time_since_stage") {
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - rule.trigger.days!);

    return db.query.proposals.findMany({
      where: and(
        eq(proposals.status, rule.trigger.stage!),
        lt(proposals.updatedAt, cutoff)
      ),
    });
  }

  if (rule.trigger.type === "engagement_signal") {
    const allProposals = await db.query.proposals.findMany({
      where: eq(proposals.status, "viewed"),
    });

    const matching = [];
    for (const proposal of allProposals) {
      const signals = await calculateEngagementSignals(proposal.id);
      if (signals[rule.trigger.signal as keyof typeof signals]) {
        matching.push(proposal);
      }
    }
    return matching;
  }

  return [];
}

async function executeAction(
  action: AutomationRule["action"],
  proposal: Proposal
) {
  switch (action.type) {
    case "send_email":
      await sendEmail({
        to: proposal.prospect.contactEmail!,
        template: action.template!,
        data: {
          proposalUrl: `${process.env.APP_URL}/p/${proposal.token}`,
          companyName: proposal.prospect.companyName,
        },
      });
      break;

    case "notify_agency":
      await notifyAgency({
        type: "automation_triggered",
        data: {
          message: action.message!,
          proposalId: proposal.id,
          domain: proposal.prospect.domain,
        },
      });
      break;
  }
}
```

```typescript
// components/proposals/PipelineView.tsx
"use client";

import { DragDropContext, Droppable, Draggable } from "@hello-pangea/dnd";

export function PipelineView({ proposals }: { proposals: Proposal[] }) {
  const stages = PIPELINE_STAGES.filter(s => !["onboarded", "declined", "expired"].includes(s.id));
  
  const proposalsByStage = stages.reduce((acc, stage) => {
    acc[stage.id] = proposals.filter(p => p.status === stage.id);
    return acc;
  }, {} as Record<string, Proposal[]>);

  const handleDragEnd = async (result: DropResult) => {
    if (!result.destination) return;

    const proposalId = result.draggableId;
    const newStage = result.destination.droppableId;

    await fetch("/api/proposals/stage", {
      method: "PATCH",
      body: JSON.stringify({ proposalId, status: newStage }),
    });
  };

  return (
    <DragDropContext onDragEnd={handleDragEnd}>
      <div className="flex gap-4 overflow-x-auto pb-4">
        {stages.map((stage) => (
          <Droppable key={stage.id} droppableId={stage.id}>
            {(provided) => (
              <div
                ref={provided.innerRef}
                {...provided.droppableProps}
                className="w-72 flex-shrink-0"
              >
                <div className="bg-gray-100 rounded-lg p-4">
                  <h3 className="font-semibold mb-3 flex items-center gap-2">
                    <span className={`w-2 h-2 rounded-full bg-${stage.color}-500`} />
                    {stage.label}
                    <span className="text-muted-foreground text-sm">
                      ({proposalsByStage[stage.id].length})
                    </span>
                  </h3>
                  
                  <div className="space-y-2">
                    {proposalsByStage[stage.id].map((proposal, index) => (
                      <Draggable
                        key={proposal.id}
                        draggableId={proposal.id}
                        index={index}
                      >
                        {(provided) => (
                          <div
                            ref={provided.innerRef}
                            {...provided.draggableProps}
                            {...provided.dragHandleProps}
                          >
                            <ProposalCard proposal={proposal} />
                          </div>
                        )}
                      </Draggable>
                    ))}
                    {provided.placeholder}
                  </div>
                </div>
              </div>
            )}
          </Droppable>
        ))}
      </div>
    </DragDropContext>
  );
}

function ProposalCard({ proposal }: { proposal: Proposal }) {
  const signals = proposal.engagementSignals;
  
  return (
    <div className="bg-white rounded-lg p-3 shadow-sm border">
      <div className="font-medium truncate">{proposal.prospect.domain}</div>
      <div className="text-sm text-muted-foreground">
        €{(proposal.monthlyFeeCents || 0) / 100}/mėn.
      </div>
      
      {/* Engagement signals */}
      <div className="flex gap-1 mt-2">
        {signals?.hot && <span title="Aktyvus">🔥</span>}
        {signals?.pricingFocused && <span title="Domisi kaina">💰</span>}
        {signals?.calculatedRoi && <span title="Skaičiavo ROI">📊</span>}
      </div>
      
      {/* Time in stage */}
      <div className="text-xs text-muted-foreground mt-2">
        {formatTimeInStage(proposal.updatedAt)}
      </div>
    </div>
  );
}
```

## Analytics Dashboard

```typescript
// server/features/proposals/analytics.ts
interface SalesAnalytics {
  pipelineValue: number;        // Total value of active proposals
  proposalsSent: number;        // Count sent this period
  viewRate: number;             // % of sent that were viewed
  winRate: number;              // % of closed that were won
  avgDealSize: number;          // Average monthly fee of won deals
  avgTimeToClose: number;       // Days from sent to paid
  lossReasons: Array<{ reason: string; count: number }>;
}

export async function calculateSalesAnalytics(
  workspaceId: string,
  period: { start: Date; end: Date }
): Promise<SalesAnalytics> {
  const proposals = await db.query.proposals.findMany({
    where: and(
      eq(proposals.workspaceId, workspaceId),
      gte(proposals.createdAt, period.start),
      lte(proposals.createdAt, period.end)
    ),
  });

  const sent = proposals.filter(p => p.sentAt);
  const viewed = proposals.filter(p => p.firstViewedAt);
  const won = proposals.filter(p => p.status === "paid" || p.status === "onboarded");
  const lost = proposals.filter(p => p.status === "declined");

  const pipelineActive = proposals.filter(p => 
    !["paid", "onboarded", "declined", "expired"].includes(p.status)
  );

  return {
    pipelineValue: pipelineActive.reduce((sum, p) => sum + (p.monthlyFeeCents || 0), 0) / 100,
    proposalsSent: sent.length,
    viewRate: sent.length > 0 ? (viewed.length / sent.length) * 100 : 0,
    winRate: (won.length + lost.length) > 0 
      ? (won.length / (won.length + lost.length)) * 100 
      : 0,
    avgDealSize: won.length > 0 
      ? won.reduce((sum, p) => sum + (p.monthlyFeeCents || 0), 0) / won.length / 100 
      : 0,
    avgTimeToClose: calculateAvgTimeToClose(won),
    lossReasons: aggregateLossReasons(lost),
  };
}
```

```typescript
// components/proposals/LossReasonModal.tsx
"use client";

import { useState } from "react";
import { Dialog, DialogContent, DialogHeader } from "@tevero/ui";

const LOSS_REASONS = [
  { id: "price", label: "Kaina per didelė" },
  { id: "competitor", label: "Pasirinko konkurentą" },
  { id: "timing", label: "Netinkamas laikas" },
  { id: "no_response", label: "Neatsakė" },
  { id: "internal", label: "Vidinis sprendimas" },
  { id: "other", label: "Kita" },
];

export function LossReasonModal({
  proposalId,
  onClose,
}: {
  proposalId: string;
  onClose: () => void;
}) {
  const [reason, setReason] = useState("");
  const [notes, setNotes] = useState("");

  const handleSubmit = async () => {
    await fetch(`/api/proposals/${proposalId}/decline`, {
      method: "POST",
      body: JSON.stringify({ reason, notes }),
    });
    onClose();
  };

  return (
    <Dialog open onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>Kodėl pasiūlymas nepriimtas?</DialogHeader>
        
        <div className="space-y-4">
          <div className="grid grid-cols-2 gap-2">
            {LOSS_REASONS.map((r) => (
              <button
                key={r.id}
                onClick={() => setReason(r.id)}
                className={`p-3 border rounded-lg text-left ${
                  reason === r.id ? "border-primary bg-primary/5" : ""
                }`}
              >
                {r.label}
              </button>
            ))}
          </div>

          {reason === "other" && (
            <textarea
              value={notes}
              onChange={(e) => setNotes(e.target.value)}
              placeholder="Papildoma informacija..."
              className="w-full p-3 border rounded-lg"
              rows={3}
            />
          )}

          <div className="flex justify-end gap-2">
            <button onClick={onClose} className="px-4 py-2 border rounded-lg">
              Atšaukti
            </button>
            <button
              onClick={handleSubmit}
              disabled={!reason}
              className="px-4 py-2 bg-primary text-white rounded-lg"
            >
              Išsaugoti
            </button>
          </div>
        </div>
      </DialogContent>
    </Dialog>
  );
}
```
